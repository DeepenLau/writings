``` js
function constfuncs() {
	var funcs = []
	for(var i = 0; i< 10; i++){
		funcs[i] = function () { return i; }
	}
	return funcs
}
// funcs[5]() => 10
```

``` js
function constfuncs() {
	var funcs = []
	for(var i = 0; i< 10; i++){
		(function(e) {
			funcs[e] = function() { return e; }
		})(i)
	}
	return funcs
}
// funcs[5]() => 5
```
